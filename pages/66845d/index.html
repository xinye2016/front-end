# iview 的组件封装及使用

# 1. table 组件封装

# 1. 代码

<template>
    <div class="xinye-table">
        <div>
            <CheckboxGroup v-model="checkedCols">
                <Checkbox v-for="(item, index) in hiddenCols" :key="index" :label="item.title"></Checkbox>
            </CheckboxGroup>
        </div>
        <Table
            ref="selection"
            :stripe="stripe"
            :border="false"
            :columns="showCols"
            :data="data"
            @on-selection-change="handleSelect"
        >
            <template slot-scope="{ row, index }" v-for="item in slots" :slot="item">
                <slot :data="row" :rowIndex="index" :name="item"></slot>
            </template>
            <template slot-scope="{ row, index }" slot="action" v-if="showActions.length > 0">
                <slot v-if="customActions" :data="row" name="action"></slot>
                <div v-else>
                    <div v-if="showActions.length <= 2">
                        <my-link v-for="(item, i) in showActions" :key="i" @on-click="handleClick(item.action, row, index)">{{
                            item.name
                        }}</my-link>
                    </div>
                    <Dropdown transfer trigger="click" @on-click="handleClick($event, row, index)" v-else>
                        <a href="javascript:void(0)">
                            更多操作
                            <Icon type="ios-arrow-down"></Icon>
                        </a>
                        <DropdownMenu slot="list">
                            <my-link
                                style="display:block;text-align:center;margin:5px 0;"
                                v-for="(item, i) in showActions"
                                :key="i"
                                :name="item.action"
                                @on-click="handleClick(item.action, row, index)"
                                >{{ item.name }}</my-link
                            >
                        </DropdownMenu>
                    </Dropdown>
                </div>
            </template>
        </Table>
        <div class="page" v-if="pageShow">
            <Page
                show-total
                show-elevator
                show-sizer
                :total="total"
                :current="current"
                @on-change="handlePageChange"
                @on-page-size-change="handlePageSizeChange"
            />
        </div>
    </div>
</template>

<script>
import MyLink from '../my-link';

export default {
    name: 'xinye-table',
    props: {
        stripe: {
            type: Boolean,
            default: false,
        },
        border: {
            type: Boolean,
            default: false,
        },
        columns: {
            type: Array,
        },
        slots: {
            type: Array,
        },
        actions: {
            type: Array,
            default: () => [],
        },
        data: {
            type: Array,
        },
        pageShow: {
            type: Boolean,
            default: true,
        },
        total: {
            type: Number,
            default: 0,
        },
        showColsCount: {
            type: Number,
            default: 0,
        },
        customActions: {
            type: Boolean,
            default: false,
        },
        current: {
            type: Number,
            default: 1,
        },
    },
    components: {
        MyLink,
    },
    data() {
        return {
            currentPage: 1,
            pageSize: 10,
            checkedCols: [],
        };
    },
    computed: {
        hiddenCols() {
            if (this.showColsCount === 0) {
                return [];
            }
            let list = [...this.columns];
            if (list[0].type === 'selection') {
                list.shift();
            }
            if (list[list.length - 1].slot === 'action') {
                list.pop();
            }
            return list.slice(this.showColsCount - list.length, list.length);
        },
        showCols() {
            if (this.showColsCount === 0) {
                return this.columns;
            }
            let list = [...this.columns];
            let add = 0;
            if (list[0].type === 'selection') {
                add++;
            }
            let action = list.find((item) => item.slot === 'action');
            let selectedList = [];
            list.forEach((item) => {
                let col = this.checkedCols.find((li) => li === item.title);
                if (col) selectedList.push(item);
            });
            let newList = list.splice(0, this.showColsCount + add);
            newList = newList.concat(selectedList);
            if (action) newList.push(action);
            return newList;
        },
        showActions() {
            let list = this.actions.filter((item) => {
                return !item.hidden;
            });
            return list;
        },
    },
    methods: {
        handleClick(action, row, index) {
            this.$emit('handleAction', { action, row, index });
        },
        handlePageChange(currentPage) {
            this.currentPage = currentPage;
            this.$emit('page-change', {
                currentPage: this.currentPage,
                pageSize: this.pageSize,
            });
        },
        handlePageSizeChange(pageSize) {
            this.pageSize = pageSize;
            this.$emit('page-change', {
                currentPage: this.currentPage,
                pageSize: this.pageSize,
            });
        },
        handleSelect(selections) {
            this.$emit('on-select', selections);
        },
    },
};
</script>

# 2. 使用参数说明

  • stripe 相邻行之间显示不同背景色 false
  • border 是否显示边框 false
  • columns table 列设置 Array
  • data table 数据 Array
  • slots 需要自定义的列表项 Array
  • total 数据总数 0
  • show-cols-count 在表格中显示的项数，其他由表格上方 checkbox 来控制显示隐藏,默认值为 0，不设置或者设置为 0 则不显示 checkbox
  • on-select 表格多选后的事件 event
  • actions 表格操作按钮设置 Array
  • custom-actions 是否显示自定义的操作按钮 false
  • handleAction 按钮操作 event
  • pageShow 是否显示分页，默认显示 true
  • page-change 分页事件 event

# 2. form 组件封装

# 1. 代码

<template>
    <Form ref="formValidate" :model="formItem" :label-width="labelWidth" class="xinye-form" :inline="inline">
        <FormItem v-for="item in options" :key="item.key" :label="item.label" :prop="item.key" :rules="item.rules">
            <Input
                clearable
                v-model="formItem[item.key]"
                :placeholder="item.placeholder"
                v-if="item.type === 'input'"
                :maxLength="item.maxLength"
            ></Input>
            <Input
                clearable
                v-model="formItem[item.key]"
                :placeholder="item.placeholder"
                v-if="item.type === 'password'"
                type="password"
            ></Input>
            <InputNumber
                type="number"
                :min="0"
                v-model="formItem[item.key]"
                :placeholder="item.placeholder"
                v-if="item.type === 'number'"
                style="width: 168px;"
            ></InputNumber>
            <Select
                clearable
                filterable
                v-model="formItem[item.key]"
                :placeholder="item.placeholder"
                v-if="item.type === 'select'"
            >
                <Option v-for="obj in item.options" :value="obj.value" :key="obj.value">{{ obj.label }}</Option>
            </Select>
            <DatePicker
                clearable
                type="date"
                v-model="formItem[item.key]"
                :placeholder="item.placeholder"
                v-if="item.type === 'date'"
            ></DatePicker>
            <DatePicker
                type="daterange"
                class="daterange"
                v-model="formItem[item.key]"
                :options="daterangeOptions"
                placement="bottom-start"
                :placeholder="item.placeholder"
                v-if="item.type === 'daterange'"
            ></DatePicker>
            <RadioGroup v-model="formItem[item.key]" :placeholder="item.placeholder" v-if="item.type === 'radio'">
                <Radio v-for="obj in item.options" :label="obj.value" :key="obj.value">{{ obj.label }}</Radio>
            </RadioGroup>
            <CheckboxGroup v-model="formItem[item.key]" :placeholder="item.placeholder" v-if="item.type === 'checkbox'">
                <Checkbox v-for="obj in item.options" :label="obj.value" :key="obj.value">{{ obj.label }}</Checkbox>
            </CheckboxGroup>
            <i-switch size="large" v-model="formItem[item.key]" :placeholder="item.placeholder" v-if="item.type === 'switch'">
                <span slot="open">是</span>
                <span slot="close">否</span>
            </i-switch>
            <Input
                type="textarea"
                class="textarea"
                v-if="item.type === 'textarea'"
                v-model="formItem[item.key]"
                :placeholder="item.placeholder"
                :autosize="item.autosize"
                :maxLength="item.maxLength"
            ></Input>
            <slot v-if="item.type === 'slot'" :name="item.key" :value="formItem[item.key]"></slot>
            <Button v-if="item.type === 'button'" type="primary" @click="handleSearch">{{ item.text }}</Button>
        </FormItem>
    </Form>
</template>
<script>
export default {
    props: {
        formItem: {
            type: Object,
            default: () => {},
        },
        options: {
            type: Array,
            default: () => [],
        },
        inline: {
            type: Boolean,
            default: false,
        },
        labelWidth: {
            type: Number,
            default: 80,
        },
    },
    data() {
        return {
            daterangeOptions: {
                shortcuts: [
                    {
                        text: '1周',
                        value() {
                            const end = new Date();
                            const start = new Date();
                            start.setTime(start.getTime() - 3600 * 1000 * 24 * 7);
                            return [start, end];
                        },
                    },
                    {
                        text: '1个月',
                        value() {
                            const end = new Date();
                            const start = new Date();
                            start.setTime(start.getTime() - 3600 * 1000 * 24 * 30);
                            return [start, end];
                        },
                    },
                    {
                        text: '3个月',
                        value() {
                            const end = new Date();
                            const start = new Date();
                            start.setTime(start.getTime() - 3600 * 1000 * 24 * 90);
                            return [start, end];
                        },
                    },
                ],
            },
        };
    },
    methods: {
        handleSearch() {
            this.$emit('on-search', this.formItem);
        },
        submit() {
            this.$refs.formValidate.validate((valid) => {
                if (valid) {
                    this.$emit('on-submit');
                }
            });
        },
        reset() {
            this.$refs.formValidate.resetFields();
        },
    },
};
</script>

<template>
    <Form :model="detail" :label-width="labelWidth" class="xinye-detail" inline>
        <FormItem v-for="item in options" :key="item.key" :label="item.label" :style="{ width: item.isLine ? '100%' : '45%' }">
            <slot :name="item.key" :value="detail[item.key]">
                <span>{{ detail[item.key] }}</span>
            </slot>
        </FormItem>
    </Form>

export default {

            type: Array,
            default: () => [],

# 2. 使用参数说明

  • detail 详情数据

  • label-width label 宽度设置，默认为 80

  • options 详情展示设置

    • key 关键字，与 detail 对象中 key 的要相同,
    • label label 值

    当要设置自定义的选项时，只需使用 slot 对应相应 key 即可

    <template #address="{ value }">
    <div style="color:red;">{{ value }}</div>
</template>
