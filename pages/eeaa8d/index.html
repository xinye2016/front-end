# 1. vue + typescript 基础

  1. MyPage
<template>
    <div>
        {{ msg }}
        <input type="button" @click="changeMsg" value="点击" />
        <p>数量: {{ doubleCount }}</p>
        <div>
            双向绑定
            <input type="text" v-model="inputMsg" />
        </div>
        <component-a :msg="inputMsg" @onClick="clickA"></component-a>
    </div>
</template>

<script lang="ts">
import { Component, Vue, Emit } from 'vue-property-decorator';
import ComponentA from '@/components/ComponentA.vue';

@Component({
    // 引入组件
    components: {
        ComponentA,
    },
})
export default class MyPage extends Vue {
    // data
    msg = 'Hello TS';
    count = 1;
    inputMsg = '132';

    // compute
    private get doubleCount(): Number {
        return this.count * 2;
    }

    // methods
    public changeMsg(): void {
        this.msg = 'Hello Vue';
    }

    public clickA(value: String): void {
        console.log(value);
    }
}
</script>

  1. Components
<template>
    <div>
        <p>{{ msg }}</p>
        <p>{{ componentMsg }}</p>
        <input type="button" value="点击子组件" @click="handleClick" />
    </div>
</template>

<script lang="ts">
import { Vue, Prop, Component, Watch, Emit } from 'vue-property-decorator';

@Component
export default class ComponentA extends Vue {
    // props
    @Prop({ default: '组件' }) readonly msg!: String;
    @Prop(String) readonly msg1!: String | undefined;
    @Prop([String, Number]) readonly msg2!: String | Number | undefined;

    // data
    componentMsg = 'Hello ComponentA';

    // $emit
    @Emit()
    handleClick() {
        this.$emit('onClick', 'Hello Page');
    }

    // watch
    @Watch('msg', { immediate: true })
    onMsgChange(value1: String, value2: String) {
        console.log(value1, value2);
    }
}
</script>

# 2. 第三方库引入

vue.config.js

'use strict';
const path = require('path');

function resolve(dir) {
    return path.join(__dirname, dir);
}

const tsImportPluginFactory = require('ts-import-plugin');

module.exports = {
    chainWebpack: (config) => {
        const tsRule = config.module.rule('ts');

        tsRule.uses.clear();

        tsRule
            .test(/\.(jsx|tsx|js|ts)$/)
            .exclude.add(resolve('node_modules'))
            .end()
            .use('ts-loader')
            .loader('ts-loader')
            .options({
                transpileOnly: true,
                getCustomTransformers: () => ({
                    before: [
                        tsImportPluginFactory({
                            libraryName: 'vant',
                            libraryDirectory: 'lib',
                            style: true,
                        }),
                    ],
                }),
                compilerOptions: {
                    module: 'es2015',
                },
            });
    },
};
