# element-ui 的组件封装及使用

# 1. table 的封装

# 1.代码

<template>
    <div class="element-table">
        <el-checkbox-group v-model="checkList" class="check-box-group">
            <el-checkbox v-for="(item, index) in hiddenCols" :key="index" :label="item.title" />
        </el-checkbox-group>
        <el-table
            :data="tableData"
            style="width: 100%"
            :stripe="tableOption.showStripe"
            :border="tableOption.showBorder"
            :size="tableOption.tableSize"
            :show-header="tableData.showHeader"
            @selection-change="handleSelectionChange"
        >
            <el-table-column v-if="tableOption.showIndex" type="index" width="50" />
            <el-table-column v-if="tableOption.showCheckbox" type="selection" width="55" :selectable="tableOption.selectable" />
            <el-table-column
                v-for="(item, index) in showColumns"
                :key="index"
                :prop="item.key"
                :label="item.title"
                :width="item.width"
                :min-width="item.minWidth"
                :show-overflow-tooltip="item.tooltip"
                :align="item.align"
                :fixed="item.fixed"
            >
                <template v-slot="scope">
                    <!-- solt 自定义列-->
                    <template v-if="item.slot">
                        <slot :name="item.key" :row="scope.row" :index="scope.$index" />
                    </template>
                    <span v-else>{{ scope.row[item.key] }}</span>
                </template>
            </el-table-column>
        </el-table>
        <div v-if="tableOption.showPage" class="page-div">
            <el-pagination
                :current-page="currentPage"
                :page-sizes="[10, 20, 30, 40]"
                :page-size="pageSize"
                layout="total, sizes, prev, pager, next, jumper"
                :total="total"
                @size-change="handleSizeChange"
                @current-change="handleCurrentChange"
            />
        </div>
    </div>
</template>

<script lang="ts">
import { Component, Prop, Vue, Watch } from 'vue-property-decorator';

interface Column {
    key: string;
    title?: string;
    width?: number;
    minWidth?: number;
    tooltip?: boolean;
    align?: string;
    fixed?: string;
    hidden?: boolean;
}

interface TableOptions {
    fixedHeader?: boolean;
    showBorder?: boolean;
    showCheckbox?: boolean;
    showHeader?: boolean;
    showIndex?: boolean;
    showStripe?: boolean;
    tableSize?: string;
}

@Component
export default class ElementTable extends Vue {
    @Prop({ default: () => [] }) readonly columns!: Column[];
    @Prop({ default: () => [] }) readonly tableData!: object[];
    @Prop({ default: () => ({}) }) readonly tableOption!: TableOptions;
    @Prop({ default: 1 }) readonly pageNum!: number;
    @Prop({ default: 0 }) readonly total!: number;

    private checkList = [];

    private currentPage = 0;
    private pageSize = 10;

    private get hiddenCols() {
        let list = this.columns.filter((item: Column) => item.hidden);
        return list;
    }

    private get showColumns() {
        let list: Column[] = [];
        this.columns.forEach((col) => {
            let index1 = this.hiddenCols.findIndex((item) => item.title === col.title);
            let index2 = this.checkList.findIndex((item) => item === col.title);
            if (index1 === -1 || index2 !== -1) {
                list.push(col);
            }
        });
        return list;
    }

    @Watch('pageNum', { immediate: true })
    pageNumChange(value: number) {
        this.currentPage = value;
    }

    private handleSizeChange(value: number) {
        this.pageSize = value;
        this.$emit('pageChange', {
            pageSize: this.pageSize,
            pageNum: this.currentPage,
        });
    }

    private handleCurrentChange(value: number) {
        this.currentPage = value;
        this.$emit('pageChange', {
            pageSize: this.pageSize,
            pageNum: this.currentPage,
        });
    }

    private handleSelectionChange(list: object[]) {
        this.$emit('selection-change', list);
    }
}
</script>

# 2. 使用参数说明

  • tableOption 表格配置
    • showBorder 是否显示纵向边框
    • showCheckbox 是否显示勾选列
      • selectable 当前行是否可以选中function
    • showHeader 是否显示表头
    • showIndex 是否显示编号
    • showStripe 相邻行之间显示不同背景色boolean
    • tableSize 表格大小 medium / small / mini
  • columns 表格配置array
    • key 关键字
    • title 列名
    • width 列宽
    • minWidth 最小列宽
    • tooltip 超出宽度是否省略号
    • align 对齐方式
    • fixed 固定right、left
    • slot 自定义插槽boolean
    • hidden 是否隐藏显示当前列
  • tableData 表格数据array
  • pageNum 当前页码
  • total 数据总数
  • handleSizeChange 当前页展示个数变化处理函数
  • handleCurrentChange 当前页码变化处理函数
  • handleSelectionChange 处理表格勾选函数

# 2. form 组件的封装

# 1. 代码

<template>
    <el-form ref="form" class="element-form" :inline="inline" :model="form" :label-width="`${labelWidth}px`">
        <el-form-item
            v-for="(item, index) in formOptions"
            :key="index"
            :label="item.label"
            :prop="item.key"
            :rules="item.rules"
            :style="{ width: `${item.type !== 'daterange' ? labelWidth + 226 : labelWidth + 244}px` }"
        >
            <el-input
                v-if="item.type === 'input'"
                v-model="form[item.key]"
                suffix-icon="el-icon-pear"
                :placeholder="item.placeholder"
                :maxlength="item.maxlength"
                :disabled="item.disabled"
                clearable
            />
            <el-input
                v-if="item.type === 'password'"
                v-model="form[item.key]"
                type="password"
                :placeholder="item.placeholder"
                suffix-icon="el-icon-pear"
                :disabled="item.disabled"
                clearable
            />
            <el-input-number
                v-if="item.type === 'inputNumber'"
                v-model="form[item.key]"
                :precision="item.precision"
                :min="item.min"
                :max="item.max"
                :placeholder="item.placeholder"
                :disabled="item.disabled"
            />
            <el-input
                v-if="item.type === 'textarea'"
                v-model="form[item.key]"
                type="textarea"
                :rows="item.rows"
                :maxlength="item.maxlength"
                :show-word-limit="item.showWordLimit"
                :placeholder="item.placeholder"
                :style="{ width: item.width }"
                :disabled="item.disabled"
            />
            <el-select
                v-if="item.type === 'select'"
                v-model="form[item.key]"
                :placeholder="item.placeholder"
                :disabled="item.disabled"
                clearable
            >
                <el-option v-for="option in item.options" :key="option.value" :label="option.label" :value="option.value" />
            </el-select>
            <el-date-picker
                v-if="item.type === 'date'"
                :key="form[item.key]"
                v-model="form[item.key]"
                type="date"
                :placeholder="item.placeholder"
                :disabled="item.disabled"
                clearable
            />
            <el-date-picker
                v-if="item.type === 'daterange'"
                v-model="form[item.key]"
                type="daterange"
                :value-format="item.valueFormate"
                :range-separator="item.rangeSeparator"
                :start-placeholder="item.startPlaceholder"
                :end-placeholder="item.endPlaceholder"
                style="width:243px"
                :disabled="item.disabled"
            />
            <el-date-picker
                v-if="item.type === 'datetimerange'"
                v-model="form[item.key]"
                type="datetimerange"
                :value-format="item.valueFormate"
                :range-separator="item.rangeSeparator"
                :start-placeholder="item.startPlaceholder"
                :end-placeholder="item.endPlaceholder"
                style="width:383px"
                :disabled="item.disabled"
            />
            <el-radio-group v-if="item.type === 'radio'" v-model="form[item.key]" :style="{ width: `${item.width}px` }">
                <el-radio v-for="radio in item.options" :key="radio.value" :label="radio.value" :disabled="item.disabled">
                    {{ radio.label }}
                </el-radio>
            </el-radio-group>
            <el-checkbox-group v-if="item.type === 'checkbox'" v-model="form[item.key]">
                <el-checkbox
                    v-for="checkbox in item.options"
                    :key="checkbox.value"
                    :label="checkbox.value"
                    :disabled="item.disabled"
                >
                    {{ checkbox.label }}
                </el-checkbox>
            </el-checkbox-group>
            <el-switch
                v-if="item.type === 'switch'"
                v-model="form[item.key]"
                active-color="#13ce66"
                inactive-color="#ff4949"
                active-text="是"
                inactive-text="否"
                :disabled="item.disabled"
            />
            <el-button v-if="item.type === 'button'" v-waves type="primary" size="small" @click="handleClick">
                {{ item.buttonText }}
            </el-button>
            <slot v-if="item.type === 'slot'" :name="item.key" />
        </el-form-item>
    </el-form>
</template>

<script lang="ts">
import { Component, Prop, Vue } from 'vue-property-decorator';
import { Form as ElForm } from 'element-ui';

interface FormOption {
    key: string;
    label: string;
    type: string;
    placeholder: string;
}

@Component
export default class ElementForm extends Vue {
    @Prop({ default: 80 }) readonly labelWidth!: number;
    @Prop({ default: false }) readonly inline!: boolean;
    @Prop({ default: () => ({}) }) readonly form!: object;
    @Prop({ default: () => [] }) readonly formOptions!: Array<FormOption>;

    private handleClick() {
        this.$emit('on-click');
    }

    public submitForm() {
        (this.$refs.form as ElForm).validate((valid: boolean) => {
            if (valid) {
                this.$emit('validate-success');
            } else {
                console.log('error submit!!');
                return false;
            }
        });
    }

    public resetForm() {
        (this.$refs.form as ElForm).resetFields();
    }
}
</script>

# 2. 使用参数说明

  • labelWidth 字段名称长度number
  • inline 是否单行boolean
  • form 表单数据object
  • formOptions 表单项array
    • key 关键字
    • label 字段名称
    • placeholder 占位字符内容
    • rules 正则验证规则array
    • type 表单项类型
      • input 普通输入框
        • maxlength 最大输入长度
      • password 密码输入框
      • inputNumber 数字输入框
        • max 最大输入值
        • min 最小输入值
        • precision 精确到小数点后的位数
      • textarea 文本输入框
        • rows 文本输入框的行数
        • maxlength 最大输入长度
        • showWordLimit 是否显示输入长度
      • select 下拉框
        • options 下拉项{value:值，label:显示文本}
      • date 日期输入框
      • daterang 日期区间输入框 datetimerange 日期区间输入框（精确到时分秒）
        • valueFormate 展示转换规则yy-MM-dd
        • rangeSeparator 区间文字
        • startPlaceholder 前区间占位文字
        • endPlaceholder 后区间占位文字
      • radio 单选框
        • options 单选项{value:值，label:显示文本}
      • checkbox 复选框
        • options 复选项{value:值，label:显示文本}
      • switch 开关
      • button 按钮
        • buttonText 按钮文字
        • handleClick 绑定函数
      • slot 自定义插槽
    • maxlength 最大长度number
    • disabled 是否禁用
    • submitForm 提交表单函数(this.$refs.form as ElForm).submitForm()
    • resetForm 清空表单函数(this.$refs.form as ElForm).resetForm()

# 3. 详情组件

# 1. 代码

<template>
    <el-form ref="form" class="element-detail" :inline="inline" :model="detail" :label-width="`${labelWidth}px`">
        <el-form-item v-for="(item, index) in showOptions" :key="index" :label="item.label" :style="{ width: item.width }">
            <slot :name="item.key" :value="detail[item.key]">
                <span>{{ detail[item.key] }}</span>
            </slot>
        </el-form-item>
    </el-form>
</template>

<script lang="ts">
import { Component, Prop, Vue } from 'vue-property-decorator';

@Component
export default class Detail extends Vue {
    @Prop({ default: () => ({}) }) readonly detail!: any;
    @Prop({ default: () => [] }) readonly options!: Array<any>;
    @Prop({ default: 80 }) readonly labelWidth!: number;
    @Prop({ default: false }) readonly inline!: boolean;

    private get showOptions() {
        let newOptions = this.options.filter((item: any) => {
            return typeof this.detail[item.key] === 'number' || Boolean(this.detail[item.key]);
        });
        return newOptions;
    }
}
</script>
